<script>

document.addEventListener('DOMContentLoaded', () => {

  const toast = document.getElementById('toast');

  function showToast(msg){toast.textContent=msg;toast.classList.add('show');setTimeout(()=>toast.classList.remove('show'),1800);}

  function collectData(){

    return {

      coach:coach_name.value,student:student_name.value,week:week.value,date:date_of_note.value,

      status:academic_status.value,responsiveness:responsiveness.value,support:support_need.value,

      risks:Array.from(document.querySelectorAll('.group input[type="checkbox"]:checked')).map(x=>x.value),

      goals:active_goals.value,summary:progress_summary.value};

  }

  function buildPlainText(d){

    const r=d.risks.length?d.risks.join('\\n'):'None';

    return`Digital AP Notes

Success Coach: ${d.coach||'—'}

Student: ${d.student||'—'}

Week: ${d.week||'—'}

Date: ${d.date||'—'}


Academic Overview

Status: ${d.status||'—'}

Responsiveness: ${d.responsiveness||'—'}

Support Need: ${d.support||'—'}


Current Risk Factors

${r}


Active Goals

${d.goals||'—'}


Progress Summary

${d.summary||'—'}


Exported ${new Date().toLocaleString()}`;}

  function downloadText(name,text){const a=document.createElement('a');a.href=URL.createObjectURL(new Blob([text]));a.download=name;document.body.append(a);a.click();a.remove();}

  async function copyText(t){try{await navigator.clipboard.writeText(t);showToast('Copied to clipboard');}catch{showToast('Copy failed');}}

  btnMarkdown.onclick=()=>{const d=collectData();downloadText(`digital_ap_notes_${(d.student||'entry').replace(/\\s+/g,'_')}.txt`,buildPlainText(d));showToast('Downloaded file');};

  btnCopyMarkdown.onclick=()=>copyText(buildPlainText(collectData()));

  btnClear.onclick=()=>{document.querySelectorAll('input,select,textarea').forEach(e=>{if(e.type==='checkbox')e.checked=false;else e.value='';});showToast('Cleared');};

});

</script>
